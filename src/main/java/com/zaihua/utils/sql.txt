CREATE TABLE k_day
(
    id SERIAL PRIMARY KEY,
    symbol CHARACTER VARYING(20) NOT NULL,
    volume CHARACTER VARYING(20) NOT NULL,
    open CHARACTER VARYING(20) NOT NULL,
    high CHARACTER VARYING(20) NOT NULL,
    close CHARACTER VARYING(20) NOT NULL,
    low CHARACTER VARYING(20) NOT NULL,
    chg CHARACTER VARYING(20) NOT NULL,
    percent CHARACTER VARYING(20) NOT NULL,
    turnrate CHARACTER VARYING(20) NOT NULL,
    ma5 CHARACTER VARYING(20) NOT NULL,
    ma10 CHARACTER VARYING(20) NOT NULL,
    ma20 CHARACTER VARYING(20) NOT NULL,
    ma30 CHARACTER VARYING(20) NOT NULL,
    dif CHARACTER VARYING(20) NOT NULL,
    dea CHARACTER VARYING(20) NOT NULL,
    macd CHARACTER VARYING(20) NOT NULL,
    time CHARACTER VARYING(20) NOT NULL,
    update_time timestamp NOT NULL ON UPDATE CURRENT_TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX index_symbol ON k_day (symbol);
CREATE INDEX index_time ON k_day (time);
CREATE UNIQUE INDEX index_symbol_time ON k_day (symbol,time);
